#define BOOST_HANA_STATIC_ASSERT (   ...)    assert((__VA_ARGS__))

Expands to a runtime assertion.

Use this only at function scope. This is a workaround because constexpr lambdas are not available yet and so our unit tests can't use static_assert.

auto boost::hana::detail::laws< Monad >
Initial value:
= [](auto monad, auto a, auto f, auto g) {
auto unit_ = unit<datatype_t<decltype(monad)>>;
return bind(unit_(a), f) == f(a) &&
bind(monad, unit_) == monad &&
bind(monad, [=](auto x) { return bind(f(x), g); }) == bind(bind(monad, f), g) &&
fmap(f, monad) == bind(monad, compose(unit_, f));
}
