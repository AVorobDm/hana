template<typename T>
struct boost::hana::Applicative::instance< T, when< is_a< List, T >()> >

Applicative instance for instances of the List type class.

A value can be lifted into a singleton list with lift. ap(fs, xs) applies each function in the list fs to each value in the list xs, and returns a list containing all the results.

Example

BOOST_HANA_STATIC_ASSERT(lift<List>('x') == list('x'));
BOOST_HANA_STATIC_ASSERT(lift<StdTuple>('x') == std::make_tuple('x'));
BOOST_HANA_CONSTEXPR_LAMBDA auto f = pair;
BOOST_HANA_CONSTEXPR_LAMBDA auto g = flip(pair);
BOOST_HANA_STATIC_ASSERT(
ap(list(f, g), list(1, 2, 3), list('a', 'b'))
==
list(
f(1, 'a'), f(1, 'b'), f(2, 'a'), f(2, 'b'), f(3, 'a'), f(3, 'b'),
g(1, 'a'), g(1, 'b'), g(2, 'a'), g(2, 'b'), g(3, 'a'), g(3, 'b')
)
);
