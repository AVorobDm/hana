template<>
struct boost::hana::Iterable::instance< Range >

Let r be a Range containing the Integrals in the half-open interval [from, to). The head of r is an Integral with value from, its tail is the range representing the [from + 1, to) interval and r is empty if and only if from == to.

BOOST_HANA_CONSTEXPR_LAMBDA auto r = range(int_<0>, int_<1000>);
BOOST_HANA_STATIC_ASSERT(head(r) == int_<0>);
BOOST_HANA_STATIC_ASSERT(last(r) == int_<999>);
BOOST_HANA_STATIC_ASSERT(!is_empty(r));
BOOST_HANA_STATIC_ASSERT(is_empty(range(int_<3>, int_<3>)));
